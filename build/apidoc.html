function decision-tree.decision_tree.prototype.evaluate (samples)

  • description and source-code
    evaluate = function (samples) {
  var instance = this;
  var target = this.target;

  var total = 0;
  var correct = 0;

  _.each(samples, function(s) {
    total++;
    var pred = instance.predict(s);
    var actual = s[target];
    if (pred == actual) {
      correct++;
    }
  });

  return correct / total;
}

function decision-tree.decision_tree.prototype.predict (sample)

  • description and source-code
    predict = function (sample) {
  var root = model;
  while (root.type !== NODE_TYPES.RESULT) {
    var attr = root.name;
    var sampleVal = sample[attr];
    var childNode = _.detect(root.vals, function(node) {
      return node.name == sampleVal
    });
    if (childNode){
      root = childNode.child;
    } else {
      root = root.vals[0].child;
    }
  }

  return root.val;
}
